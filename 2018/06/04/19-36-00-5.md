+/* Clear the environment.  */
+extern "C" int
+clearenv ()
+{
+  __try
+    {
+      if (cur_environ () == lastenviron && lastenviron != NULL)
+	{
+	  free (lastenviron);
+	  lastenviron = NULL;
+	}
+      __cygwin_environ = NULL;
+      update_envptrs ();
+      return 0;
+    }
+  __except (EFAULT) {}
+  __endtry
+  return -1;
+}
