```
  static int
  grab_int (char **p)
  {
    char *src = *p, *stp;
    int val = strtol (src, &stp, 10);
    if (stp == src || *stp != ':')
      return -1;
    *p = stp + 1;
    return val;
  }

```

static int
parse_pwd (struct passwd &res, char *buf)
{
```
  [...]
  if ((res.pw_uid = grab_int (&buf)) < 0)
    return 0;
  if ((res.pw_gid = grab_int (&buf)) < 0)
    return 0;
  [...]

```
