#include <unistd.h>

main(int argc, char **argv)

```
  struct passwd pw, *ret;
  char buffer[256];

  if (getpwuid_r (100, &pw, buffer, 256, &ret))
    perror ("getpwuid_r");
  else
    {
      printf ("pw_name: <%s>\n", pw.pw_name);
      printf ("pw_dir: <%s>\n", pw.pw_dir);
      printf ("pw_shell: <%s>\n", pw.pw_shell);
    }
  return 0;
```

}
